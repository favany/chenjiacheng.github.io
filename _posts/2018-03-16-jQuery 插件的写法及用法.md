## jQuery 扩展插件的格式

```
(function($){...})(jQuery);

// 当然，为了防止压缩合并等操作时，它前面的js代码没有结束分号而引起错误，这里可以在前面加个;就是：

;(function($){...})(jQuery);

// 传入参数可以有多个，比如：

;(function($, window,document){...})(jQuery,window, document);
```

## jQuery 元素方法的扩展

```
$.fn.extend({
    myMethod: function () {
        // 用this可以拿到调用该方法的jq dom元素
    }
});
```

或者

```
$.fn.myMethod = function() {
    // 用this可以拿到调用该方法的jq dom元素
};
```

## jQuery 滚动加载数据插件示例

```
;(function ($) {
    let config = {
            url: null,
            type: 'post',
            dataType: 'json',
            data: {},
            success: function (res) {
            },
            page: 1,
            limit: 10,
            loadingText: '加载中...',
            loadendText: '没有数据了',
            errorText: '加载失败',
        },
        status = false,
        data = {
            page: config.page,
            limit: config.limit,
        },
        getLoadingTpl = function ($data) {
            return '<div class="wl-loadmore" style="font-size: 14px;color: #999;text-align: center;padding: 10px 0;">' + $data + '</div>';
        },
        getLoadendTpl = function ($data) {
            return '<div class="wl-loadmore" style="font-size: 14px;color: #999;text-align: center;padding: 10px 0;">' + $data + '</div>';
        },
        getLoadData = function (obj) {
            status = true;
            if (!config.url) {
                console.error('请配置对应的url');
                return;
            }
            $.ajax({
                url: config.url,
                type: config.type,
                dataType: config.dataType,
                data: $.extend(data, config.data),
                beforeSend: function () {
                    $(obj).append(getLoadingTpl(config.loadingText));
                },
                success: function (result) {
                    $(obj).children('.wl-loadmore').remove();
                    if (this.dataType === 'html') {
                        config.success(result);
                        if (result === '') {
                            $(obj).append(getLoadendTpl(config.loadendText));
                        } else {
                            status = false;
                            data.page++;
                        }
                    } else {
                        config.success(result.data);
                        if (result.current_page >= result.last_page) {
                            $(obj).append(getLoadendTpl(config.loadendText));
                        } else {
                            status = false;
                            data.page++;
                        }
                    }
                },
                error: function () {
                    $(obj).children('.wl-loadmore').remove();
                    $(obj).append(getLoadendTpl(config.errorText));
                }
            });
        },
        init = function (obj) {
            getLoadData(obj);
            $(window).scroll(function () {
                if ($(window).scrollTop() + 10 >= $(document).height() - $(window).height()) {
                    if (status) return;
                    status = true;
                    getLoadData(obj);
                }
            });
        };
    $.fn.more = function (param) {
        if (param) {
            $.extend(config, param);
            data = {
                page: config.page,
                limit: config.limit,
            }
        }
        init(this);
        return this;
    };
    $.fn.reload = function (container, requestParams) {
        container.empty();
        config.page = 1;
        data = {
            page: config.page,
            limit: config.limit,
        };
        config.data = $.extend(config.data, requestParams);
        getLoadData(this);
    }
})(jQuery);
```
