## 需求背景

先看一个实际的需求，有一个 15 × 15 的五子棋盘中，实现存盘退出和续上盘的功能。如下图所示，你会怎么做？

![01.png](/static/images/20210809/01.png)

## 普通数组方法实现

将棋盘抽象化，用一个 15 × 15 的数组来表示，0 表示空点，1 表示黑子，2 表示白子。

### 代码实现

```go
package main

import "fmt"

func main() {
	var chessMap [15][15]int

	// 黑子
	chessMap[2][2] = 1

	// 白子
	chessMap[3][3] = 2

	// 打印结果
	for _, v := range chessMap {
		for _, v2 := range v {
			fmt.Print(v2)
		}
		fmt.Println()
	}
}
```

### 不足之处

因为该二维数组的很多值是默认值 0，因此记录了很多没有意义的数据。将这些没有意义的数据存储起来，不但会浪费存储空间的大小，如果写入到磁盘中，还会增大 IO 的读写量，影响性能。

## 稀疏数组方法实现

当一个数组中大部分元素为０，或者为同一个值的数组时，可以使用稀疏数组来保存该数组。

1. 记录数组一共有几行几列，有多少个不同的值。

2. 思想：把具有不同值的元素的行列及值记录在一个小规模的数组中，从而缩小程序的规模。

### 代码实现

1. 使用稀疏数组，来保留类似前面的二维数组(棋盘、地图等等)。

2. 把稀疏数组存盘，并且可以从新恢复原来的二维数组数。

```go
package main

import (
	"fmt"
)

type ValNode struct {
	row int
	col int
	val int
}

func main() {

	/*----------------------------原始数组转稀疏数组----------------------------*/

	// 1. 先创建一个原始数组
	var chessMap [15][15]int
	chessMap[2][2] = 1
	chessMap[3][3] = 2

	// 2. 输出查看原始的数组
	for _, v := range chessMap {
		for _, v2 := range v {
			fmt.Print(v2)
		}
		fmt.Println()
	}
	/*输出结果：
	000000000000000
	000000000000000
	001000000000000
	000200000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000*/

	// 3. 转成稀疏数组
	var sparseArr []ValNode
	valNode := ValNode{
		row: 15,
		col: 15,
		val: 0,
	}
	sparseArr = append(sparseArr, valNode)

	for i, v := range chessMap {
		for j, v2 := range v {
			if v2 != 0 {
				valNode := ValNode{
					row: i,
					col: j,
					val: v2,
				}
				sparseArr = append(sparseArr, valNode)
			}
		}
	}

	// 4. 输出稀疏数组
	for _, node := range sparseArr {
		fmt.Println(node.row, node.col, node.val)
	}
	/*输出结果：
	15 15 0
	2 2 1
	3 3 2*/

	/*----------------------------稀疏数组转原始数组----------------------------*/

	// 1. 先创建一个原始数组
	var chessMap2 [15][15]int

	// 2. 转成原始数组
	for i, node := range sparseArr {
		if i == 0 {
			continue
		}
		chessMap2[node.row][node.col] = node.val
	}

	// 3. 输出原始数组
	for _, v := range chessMap2 {
		for _, v2 := range v {
			fmt.Print(v2)
		}
		fmt.Println()
	}
	/*输出结果：
	000000000000000
	000000000000000
	001000000000000
	000200000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000*/
}
```

### 优化代码

1. 坑点一：读取后是字符串，需要分割成字符串数组，这里通过 Split 但是**数组最后一个多一个换行符号需要去掉**。
2. 坑点二：**Go 是无法直接在二维数组创建时传变量的**。这个时候先声明一个二维切片。初始化二维切片后。填入初始值。

```go
package main

import (
	"bufio"
	errors "errors"
	"fmt"
	"io"
	"os"
	"strconv"
	"strings"
)

type ValNode struct {
	row int
	col int
	val int
}

// SaveChessMap 保存棋谱
func SaveChessMap(filePath string, chessMap [15][15]int) error {

	// 打开文件
	file, err := os.OpenFile(filePath, os.O_WRONLY|os.O_CREATE, 0666)
	if err != nil {
		return errors.New("打开文件错误!")
	}
	defer file.Close()

	var sparseArr []ValNode
	sparseArr = append(sparseArr, ValNode{
		row: 15,
		col: 15,
		val: 0,
	})

	for i, v := range chessMap {
		for j, v2 := range v {
			if v2 != 0 {
				sparseArr = append(sparseArr, ValNode{
					row: i,
					col: j,
					val: v2,
				})
			}
		}
	}

	// 写入文件
	writer := bufio.NewWriter(file)
	for _, node := range sparseArr {
		str := fmt.Sprintf("%d %d %d\n", node.row, node.col, node.val)
		writer.WriteString(str)
	}

	writer.Flush()

	return nil
}

// ReadChessMap 读取棋谱
func ReadChessMap(filePath string) [][]int {
	file, err := os.Open(filePath)
	if err != nil {
		fmt.Println(err)
		os.Exit(2)
	}

	defer file.Close()

	reader := bufio.NewReader(file)

	// 获取第一行数据
	str, _ := reader.ReadString('\n')
	// 通过空格分割字符串
	arr := strings.Split(str, " ")
	m, err := strconv.Atoi(arr[0])
	n, err := strconv.Atoi(arr[1])
	v, err := strconv.Atoi(strings.Replace(arr[2], "\n", "", -1))
	if err != nil {
		fmt.Println(err)
		os.Exit(2)
	}

	// Go 是无法直接在二维数组创建时传变量的 var chessMap [m][n]int 不能成功
	var chessMap [][]int
	for i := 0; i < m; i++ {
		chessMap = append(chessMap, make([]int, n)) // 把一维切片，当作一个整体传入二维切片中
	}
	for i := 0; i < m; i++ {
		for j := 0; j < n; j++ {
			chessMap[i][j] = v
		}
	}

	// 循环的读取文件的内容
	for {
		str, err := reader.ReadString('\n')
		// io.EOF 表示文件的末尾
		if err == io.EOF {
			break
		}

		arr := strings.Split(str, " ")
		m, err := strconv.Atoi(arr[0])
		n, err := strconv.Atoi(arr[1])
		v, err := strconv.Atoi(strings.Replace(arr[2], "\n", "", -1))
		if err != nil {
			fmt.Println(err)
			os.Exit(2)
		}

		chessMap[m][n] = v

	}

	return chessMap
}

func main() {

	/*----------------------------保存棋谱----------------------------*/

	var chessMap [15][15]int
	chessMap[2][2] = 1
	chessMap[3][3] = 2

	err := SaveChessMap("D:/chessmap.data", chessMap)
	if err != nil {
		fmt.Println("存储失败")
	} else {
		fmt.Println("存储成功")
	}

	/*----------------------------读取棋谱----------------------------*/

	chessMap2 := ReadChessMap("D:/chessmap.data")
	for _, v := range chessMap2 {
		for _, v2 := range v {
			fmt.Print(v2)
		}
		fmt.Println()
	}
	/*输出结果：
	000000000000000
	000000000000000
	001000000000000
	000200000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000
	000000000000000*/
}
```
